389 char SpectrogramVisualizer::colorByteMap(float x) {
390  double fac = 20.0 * colorScale[1]; // colorScale[1] units: intensity/dB
391  auto k = (int) (colorScale[0] + fac * log10((double)x));
392  if (k > 255) k = 255; else if (k < 0) k = 0;
393 
394  if (colorMode == 1) {
395  // inverse b/w
396  return (char) (255 - k);
397  } else if (colorMode == 2) { // color (pack into 3_3_2 RGB format) .. SLOW?
398  float a = k / 255.0f; // 0<a<1. now map from [0,1] to rgb in [0,1]
399  float r = 5 * (a - 0.2);
400  if (r < 0) r = 0.0; else if (r >= 1) r = .955; // clip
401  float g = 5 * (a - 0.6);
402  if (g < 0) g = 0.0; else if (g >= 1) g = .995;
403  float b = 5 * a;
404  if (a > 0.8) b = 5 * (a - 0.8); else if (a > 0.4) b = 5 * (0.6 - a);
405  if (b < 0) b = 0.0; else if (b >= 1) b = .995;
406  return (char) (b * 4 + 4 * ((int) (g * 8)) + 32 * ((int) (r * 8))); // pack to 8-bits
407  } else {
408  // b/w
409  return (char) k;
410  }
411 }

420 int SpectrogramVisualizer::chooseTics(float lowValue, float range, float fudgeFactor, float *tickMarks) {
421  int i, nTics, startTick;
422  float exponent, logr, spacing;
423  if (fudgeFactor == 0.0) fudgeFactor = 1.0;
424 
425  /* adjust the range multiplier here to give good tick density */
426  logr = (float) log10(range * 0.2 / fudgeFactor); // 0.4
427  exponent = floor(logr);
428  spacing = (float) pow(10.0, exponent);
429  if (logr - exponent > log10(5.0))
430  spacing *= 5.0;
431  else if (logr - exponent > log10(2.0))
432  spacing *= 2.0;
433 
434  /* (int) and copy-sign trick is to convert the floor val to an int */
435  startTick = (int) (copysign(0.5, lowValue) + 1.0 + floor(lowValue / spacing));
436 
437  nTics = (int) (1.0 + (lowValue + range - startTick * spacing) / spacing);
438  for (i = 0; i < nTics; ++i) {
439  tickMarks[i] = spacing * (startTick + i);
440  }
441  return nTics;
442 }

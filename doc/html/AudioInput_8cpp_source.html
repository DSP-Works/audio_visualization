99 void AudioInput::initializeWindow(float *window, int windowSize, unsigned int windowType) {
100  float W;
101  int i;
102 
103  switch (windowType) {
104  case 0:
105  /* no window (crappy frequency spillover) */
106  for (i = 0; i < windowSize; ++i)
107  window[i] = 1.0F;
108  break;
109  case 1:
110  /* Hann window (C^1 cont, so third-order tails) */
111  W = windowSize / 2.0F;
112  for (i = 0; i < windowSize; ++i)
113  window[i] = (float) (1.0f + cos(M_PI * (i - W) / W)) / 2;
114  break;
115  case 2:
116  /* truncated Gaussian window (Gaussian tails + exp small error) */
117  /* width: keep small truncation but wide to not waste FFT */
118  W = windowSize / 5.0F;
119  for (i = 0; i < windowSize; ++i) {
120  window[i] = (float) exp(-(i - windowSize / 2) * (i - windowSize / 2) / (2 * W * W));
121  }
122  break;
123  default:
124  fprintf(stderr, "unknown windowType!\n");
125  }
126 }

128 void AudioInput::computeSpectrogramSlice(AudioInput *audioInput) {
129  int nfft = audioInput->fftLength; // transform length
130  int nf = audioInput->N_FREQUENCIES; // # freqs to fill in powerspec
131  int ind;
132 
133  /* copy the nfft most recent samples & multiply by the window */
134  //Log::getInstance()->logger() << "BufferSizeSamples: " << audioInput->bufferSizeSamples << std::endl;
135  for (int i = 0; i < nfft; ++i) {
136  ind = mod(audioInput->bufferIndex - nfft + i, audioInput->bufferSizeSamples);
137  audioInput->windowedAudioFrame[i] = audioInput->windowingFunction[i] * audioInput->audioBuffer[ind];
138  }
139 
140  /* execute the configured FFT */
141  fftwf_execute(audioInput->fftPlan);
142 
143  if (nf > nfft / 2) {
144  fprintf(stderr, "window too short cf n_f!\n");
145  return;
146  }
147 
148  /* zero-frequency has no imaginary part */
149  audioInput->spectrogramSlice[0] = audioInput->fftFrame[0] * audioInput->fftFrame[0];
150 
151  /* compute power spectrum from hc dft */
152  for (int i = 1; i < nf; ++i) {
153  audioInput->spectrogramSlice[i] =
154  audioInput->fftFrame[i] * audioInput->fftFrame[i] +
155  audioInput->fftFrame[nfft - i] * audioInput->fftFrame[nfft - i];
156  }
157 }

68 void Display::display()
69 {
70  glClear(GL_COLOR_BUFFER_BIT); // no depth buffer
71  glMatrixMode(GL_PROJECTION);
72  glLoadIdentity();
73  glOrtho(0, 1, 0, 1, -1, 1); // l r b t n f
74  glMatrixMode(GL_MODELVIEW);
75  glLoadIdentity();
76  std::for_each(graphicsItems.begin(), graphicsItems.end(), [&](auto item) { item->display(); });
77  glFinish(); // wait for all gl commands to complete
78  glutSwapBuffers(); // for this to WAIT for vSync, need enable in NVIDIA OpenGL
79 }
